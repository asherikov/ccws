92  class WriterWrapper
93  {
94  public:
95  std::mutex mutex_;
96  const std::string id_;
97  ariles2::namevalue2::Writer::Parameters writer_parameters_;
98  std::shared_ptr<NameValueContainer> data_;
99  ariles2::namevalue2::Writer writer_;
100 
101  bool published_;
102 
103 
104  public:
105  WriterWrapper(
106  const ariles2::DefaultBase &source,
107  std::string id,
108  const bool persistent_structure,
109  uint32_t &names_version)
110  : id_(std::move(id)), data_(std::make_shared<NameValueContainer>()), writer_(data_)
111  {
112  writer_parameters_ = writer_.getDefaultParameters();
113  if (persistent_structure)
114  {
115  writer_parameters_.persistent_structure_ = true;
116  }
117 
118  // write to allocate memory
119  ariles2::apply(writer_, source, id_);
120  data_->finalize(writer_parameters_.persistent_structure_, rclcpp::Time(0), names_version);
121 
122  published_ = true; // do not publish on assignment
123  }
124 
125 
126  void publish(const NamesPublisherPtr &names_sink, const ValuesPublisherPtr &values_sink)
127  {
128  if (mutex_.try_lock())
129  {
130  if (not published_)
131  {
132  if (data_->new_names_version_)
133  {
134  names_sink->publish(data_->names_);
135  data_->new_names_version_ = false;
136  }
137  values_sink->publish(data_->values_);
138  published_ = true;
139  }
140  mutex_.unlock();
141  }
142  }
143 
144 
145  void write(const ariles2::DefaultBase &source, const rclcpp::Time &timestamp, uint32_t &names_version)
146  {
147  if (mutex_.try_lock())
148  {
149  ariles2::apply(writer_, source, id_);
150  data_->finalize(writer_parameters_.persistent_structure_, timestamp, names_version);
151  published_ = false;
152 
153  mutex_.unlock();
154  }
155  }
156  };

238  Implementation(const std::string &sink_id, const std::size_t rate)
239  {
240  names_version_ = intrometry::backend::getRandomUInt32();
241 
242  const std::string node_id = intrometry::backend::normalizeId(sink_id);
243  const std::string random_id = intrometry::backend::getRandomId(8);
244  const std::string topic_prefix =
245  intrometry::backend::str_concat("intrometry/", node_id.empty() ? random_id : node_id);
246 
247  node_ = std::make_shared<rclcpp::Node>(
248  intrometry::backend::str_concat("intrometry_", node_id, "_", random_id),
249  // try to be stealthy and use minimal resources
250  rclcpp::NodeOptions()
251  .enable_topic_statistics(false)
252  .start_parameter_services(false)
253  .start_parameter_event_publisher(false)
254  .append_parameter_override("use_sim_time", false)
255  .use_clock_thread(false)
256  .enable_rosout(false));
257  thread_supervisor_.initializeLogger(node_);
258 
259  names_publisher_ = node_->create_publisher<NamesMsg>(
260  intrometry::backend::str_concat(topic_prefix, "/names"),
261  rclcpp::QoS(/*history_depth=*/20).reliable().transient_local());
262  values_publisher_ = node_->create_publisher<ValuesMsg>(
263  intrometry::backend::str_concat(topic_prefix, "/values"),
264  rclcpp::QoS(/*history_depth=*/20).best_effort().durability_volatile());
265 
266 
267  thread_supervisor_.add(
268  tut::thread::Parameters(
269  tut::thread::Parameters::Restart(/*attempts=*/100, /*sleep_ms=*/50),
270  tut::thread::Parameters::TerminationPolicy::IGNORE,
271  tut::thread::Parameters::ExceptionPolicy::CATCH),
272  &Implementation::spin,
273  this,
274  rate);
275  }

280  }

283  void spin(const std::size_t rate)
284  {
285  intrometry::backend::RateTimer timer(rate);
286 
287  if (timer.valid())
288  {
289  timer.start();
290 
291  while (rclcpp::ok() and not thread_supervisor_.isInterrupted())
292  {
293  sources_.tryVisit([this](WriterWrapper &writer)
294  { writer.publish(names_publisher_, values_publisher_); });
295  rclcpp::spin_some(node_);
296 
297  timer.step();
298  }
299  }
300  else
301  {
302  thread_supervisor_.log("Incorrect spin rate");
303  }
304  thread_supervisor_.interrupt();
305  }
