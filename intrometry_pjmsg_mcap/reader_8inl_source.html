654 Status McapReader::ReadRecord(IReadable& reader, uint64_t offset, Record* record) {
655  // Check that we can read at least 9 bytes (opcode + length)
656  auto maxSize = reader.size() - offset;
657  if (maxSize < 9) {
658  const auto msg =
659  internal::StrCat("cannot read record at offset ", offset, ", ", maxSize, " bytes remaining");
660  return Status{StatusCode::InvalidFile, msg};
661  }
662 
663  // Read opcode and length
664  std::byte* data;
665  uint64_t bytesRead = reader.read(&data, offset, 9);
666  if (bytesRead != 9) {
667  return StatusCode::ReadFailed;
668  }
669 
670  // Parse opcode and length
671  record->opcode = OpCode(data[0]);
672  record->dataSize = internal::ParseUint64(data + 1);
673 
674  // Read payload
675  maxSize -= 9;
676  if (maxSize < record->dataSize) {
677  const auto msg = internal::StrCat("record type 0x", internal::ToHex(uint8_t(record->opcode)),
678  " at offset ", offset, " has length ", record->dataSize,
679  " but only ", maxSize, " bytes remaining");
680  return Status{StatusCode::InvalidRecord, msg};
681  }
682  bytesRead = reader.read(&record->data, offset + 9, record->dataSize);
683  if (bytesRead != record->dataSize) {
684  const auto msg =
685  internal::StrCat("attempted to read ", record->dataSize, " bytes for record type 0x",
686  internal::ToHex(uint8_t(record->opcode)), " at offset ", offset,
687  " but only read ", bytesRead, " bytes");
688  return Status{StatusCode::ReadFailed, msg};
689  }

694 Status McapReader::ReadFooter(IReadable& reader, uint64_t offset, Footer* footer) {
695  std::byte* data;

699  }

1802  const std::function<void(const Message&, RecordOffset)> onMessage)
1803  : mcapReader_(reader)
1804  , recordReader_(*mcapReader_.dataSource(), 0, 0)
1805  , options_(options)
1806  , onMessage_(onMessage)
1807  , queue_(options_.readOrder == ReadMessageOptions::ReadOrder::ReverseLogTimeOrder) {
1808  auto chunkIndexes = mcapReader_.chunkIndexes();
1809  if (chunkIndexes.size() == 0) {
1810  status_ = mcapReader_.readSummary(ReadSummaryMethod::AllowFallbackScan);
1811  if (!status_.ok()) {
1812  return;
1813  }
1814  chunkIndexes = mcapReader_.chunkIndexes();
1815  }
1816  if (chunkIndexes.size() == 0 || chunkIndexes[0].messageIndexLength == 0) {
1817  status_ = Status(StatusCode::NoMessageIndexesAvailable,
1818  "cannot read MCAP in time order with no message indexes");
1819  return;
1820  }
1821  for (const auto& [channelId, channel] : mcapReader_.channels()) {
1822  if (!options_.topicFilter || options_.topicFilter(channel->topic)) {
1823  selectedChannels_.insert(channelId);
1824  }
1825  }
1826  // Initialize the read job queue by finding all of the chunks that need to be read from.
1827  for (const auto& chunkIndex : mcapReader_.chunkIndexes()) {
1828  if (chunkIndex.messageStartTime >= options_.endTime) {
1829  // chunk starts after requested time range, skip it.
1830  continue;
1831  }
1832  if (chunkIndex.messageEndTime < options_.startTime) {
1833  // chunk end before requested time range starts, skip it.
1834  continue;
1835  }
1836  for (const auto& channelId : selectedChannels_) {
1837  if (chunkIndex.messageIndexOffsets.find(channelId) != chunkIndex.messageIndexOffsets.end()) {
1838  internal::DecompressChunkJob job;
1839  job.chunkStartOffset = chunkIndex.chunkStartOffset;
1840  job.messageIndexEndOffset =
1841  chunkIndex.chunkStartOffset + chunkIndex.chunkLength + chunkIndex.messageIndexLength;
1842  job.messageStartTime = chunkIndex.messageStartTime;
1843  job.messageEndTime = chunkIndex.messageEndTime;
1844  queue_.push(std::move(job));
1845  break;
1846  }
1847  }
1848  }

1892 bool IndexedMessageReader::next() {
1893  while (queue_.len() != 0) {
1894  auto nextItem = queue_.pop();
1895  if (std::holds_alternative<internal::DecompressChunkJob>(nextItem)) {
1896  const auto& decompressChunkJob = std::get<internal::DecompressChunkJob>(nextItem);
1897  // The job here is to decompress the chunk into a slot, then use the message
1898  // indices after the chunk to push ReadMessageJobs onto the queue for every message
1899  // in that chunk that needs to be read.
1900 
1901  // First, find a chunk slot to decompress this chunk into.
1902  size_t chunkReaderIndex = findFreeChunkSlot();
1903  auto& chunkSlot = chunkSlots_[chunkReaderIndex];
1904  chunkSlot.chunkStartOffset = decompressChunkJob.chunkStartOffset;
1905  // Point the record reader at the chunk and message indices after it.
1906  recordReader_.reset(*mcapReader_.dataSource(), decompressChunkJob.chunkStartOffset,
1907  decompressChunkJob.messageIndexEndOffset);
1908  for (auto record = recordReader_.next(); record != std::nullopt;
1909  record = recordReader_.next()) {
1910  switch (record->opcode) {
1911  case OpCode::Chunk: {
1912  Chunk chunk;
1913  status_ = McapReader::ParseChunk(*record, &chunk);
1914  if (!status_.ok()) {
1915  return false;
1916  }
1917  decompressChunk(chunk, chunkSlot);
1918  if (!status_.ok()) {
1919  return false;
1920  }
1921  } break;
1922  case OpCode::MessageIndex: {
1923  MessageIndex messageIndex;
1924  status_ = McapReader::ParseMessageIndex(*record, &messageIndex);
1925  if (!status_.ok()) {
1926  return false;
1927  }
1928  if (selectedChannels_.find(messageIndex.channelId) != selectedChannels_.end()) {
1929  for (const auto& [timestamp, byteOffset] : messageIndex.records) {
1930  if (timestamp >= options_.startTime && timestamp < options_.endTime) {
1931  internal::ReadMessageJob job;
1932  job.chunkReaderIndex = chunkReaderIndex;
1933  job.offset.offset = byteOffset;
1934  job.offset.chunkOffset = decompressChunkJob.chunkStartOffset;
1935  job.timestamp = timestamp;
1936  queue_.push(std::move(job));
1937  chunkSlot.unreadMessages++;
1938  }
1939  }
1940  }
1941  } break;
1942  default:
1943  status_ = Status(StatusCode::InvalidRecord,
1944  internal::StrCat("expected only chunks and message indices, found ",
1945  OpCodeString(record->opcode)));
1946  return false;
1947  }
1948  }
1949  } else if (std::holds_alternative<internal::ReadMessageJob>(nextItem)) {
1950  // Read the message out of the already-decompressed chunk.

1962  return false;

1973  return false;
1974  }
1975  onMessage_(message, readMessageJob.offset);

1977  }
1978  }
