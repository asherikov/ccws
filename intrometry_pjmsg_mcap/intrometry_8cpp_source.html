131  void write(mcap::McapWriter &writer, std::vector<std::byte> &buffer, const t_Message &message)
132  {
133  buffer.resize(getSize(message));
134  message_.data = buffer.data();
135 
136  {
137  eprosima::fastcdr::FastBuffer cdr_buffer(
138  reinterpret_cast<char *>(buffer.data()), buffer.size()); // NOLINT
139  eprosima::fastcdr::Cdr ser(
140  cdr_buffer, eprosima::fastcdr::Cdr::DEFAULT_ENDIAN, eprosima::fastcdr::CdrVersion::XCDRv1);
141  ser.set_encoding_flag(eprosima::fastcdr::EncodingAlgorithmFlag::PLAIN_CDR);
142 
143  ser.serialize_encapsulation();
144  ser << message;
145  ser.set_dds_cdr_options({ 0, 0 });
146 
147  message_.dataSize = ser.get_serialized_data_length();
148  }

152 
153 
154  const mcap::Status res = writer.write(message_);
155  if (not res.ok())

206  {
207  public:
208  std::mutex mutex_;
209  const std::string id_;
210  ariles2::namevalue2::Writer::Parameters writer_parameters_;
211  std::shared_ptr<NameValueContainer> data_;
212  ariles2::namevalue2::Writer writer_;
213 
214  bool serialized_;
215 
216  public:
217  WriterWrapper(
218  const ariles2::DefaultBase &source,
219  std::string id,
220  const bool persistent_structure,
221  uint32_t &names_version)
222  : id_(std::move(id)), data_(std::make_shared<NameValueContainer>()), writer_(data_)
223  {
224  writer_parameters_ = writer_.getDefaultParameters();
225  if (persistent_structure)
226  {
227  writer_parameters_.persistent_structure_ = true;
228  }
229 
230  // write to allocate memory
231  ariles2::apply(writer_, source, id_);
232  data_->finalize(writer_parameters_.persistent_structure_, 0, names_version);
233 
234  serialized_ = true; // do not serialize on assignment
235  }
236 
237  void serialize(McapCDRWriter &mcap_writer)
238  {
239  if (mutex_.try_lock())
240  {
241  if (not serialized_)
242  {
243  if (data_->new_names_version_)
244  {
245  mcap_writer.write(data_->names_);
246  data_->new_names_version_ = false;
247  }
248 
249  mcap_writer.write(data_->values_);
250  serialized_ = true;
251  }
252  mutex_.unlock();
253  }
254  }
255 
256  void write(const ariles2::DefaultBase &source, const uint64_t timestamp, uint32_t &names_version)
257  {
258  if (mutex_.try_lock())
259  {
260  ariles2::apply(writer_, source, id_);
261  data_->finalize(writer_parameters_.persistent_structure_, timestamp, names_version);
262  serialized_ = false;
263 
264  mutex_.unlock();
265  }
266  }
267  };

362
