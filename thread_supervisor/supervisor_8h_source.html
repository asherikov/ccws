241  void startLoop(Supervisor<t_Logger> *supervisor, t_Function &&function, t_Args &&...args)
242  {
243  bool started = false;
244  for (std::size_t attempt = 0; parameters_.restart_.isOk(attempt) and not supervisor->isInterrupted();
245  ++attempt)
246  {
247  if (started)
248  {
249  parameters_.restart_.wait();
250  if (supervisor->isInterrupted())
251  {
252  break;
253  }
254 
255  supervisor->log(
256  "Supervisor / restarting thread: ",
257  attempt + 1,
258  " / ",
259  parameters_.restart_.isUnlimited() ? 0 : parameters_.restart_.attempts_);
260  }
261  else
262  {
263  started = true;
264  }

346  UNDEFINED = 0,

349  };

352  protected:
353  std::atomic<Status> status_;

358  std::mutex terminated_threads_mutex_;

360 
361 
362  protected:
363  bool wait(const std::size_t wait_ms)
364  {
365  const std::size_t sleep_ms = 10;
366 
367  {
368  const std::lock_guard<std::mutex> lock(threads_mutex_);
369  std::size_t counter = 0;
370  for (;;)
371  {
372  if (counter * sleep_ms < wait_ms)
373  {
374  {
375  const std::lock_guard<std::mutex> terminated_lock(terminated_threads_mutex_);
376 
377  for (const Thread::Reference &thread_ref : terminated_threads_)
378  {
379  thread_ref->join();
380  threads_.erase(thread_ref);
381  }
382  terminated_threads_.clear();
383  }
384 
385  if (not threads_.empty())
386  {
387  std::this_thread::sleep_for(std::chrono::milliseconds(sleep_ms));
388  ++counter;
389  }
390  else
391  {
392  return (true);
393  }
394  }
395  else
396  {
397  // cppcheck-suppress ignoredReturnValue
398  log("Threads did not terminate in the given time after request.");
399  return (false);
400  }
401  }
402  }
403  }

407  {

409  return (threads_.empty());
